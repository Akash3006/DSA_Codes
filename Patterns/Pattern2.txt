  
* * * * * * * * *
  * * * * * * *
    * * * * *
     * * *
       *

int main()
{
    int row = 5;
    for(int i =row-1 ;i>=0;i--)
      {

        //spaces
        for(int j = 0; j < row - i -1; j++)
          cout<<" ";

        //asteris
        for(int k = (2*i+1); k >=0;k--)
          cout<<"*";

      }
    return 0;
}
